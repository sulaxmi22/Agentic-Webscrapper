# utils/llm_extractor.py

import os
import json
import re
import asyncio
import traceback
from openai import OpenAI
from dotenv import load_dotenv
from schemas.article_schema import EuronArticleList

# Load environment variables
load_dotenv()

# Initialize OpenAI client (use `EURON_BASE_URL` if provided, otherwise default to euron root)
raw_base = os.getenv("EURON_BASE_URL", "https://api.euron.one/api/v1/euri")
base_url = raw_base.strip().strip('"').strip("'")
if not base_url:
    base_url = "https://api.euron.ai"
if not base_url.startswith("http"):
    base_url = "https://" + base_url
base_url = base_url.rstrip("/")
print(f"[llm_extractor] Using EURON base_url={base_url}")
client = OpenAI(
    api_key=os.getenv("OPENAI_API_KEY"),
    base_url=base_url,
    timeout=60,
)


async def process_with_llm(html: str, instructions: str, truncate: bool = False):
    """
    Send HTML content to the LLM and extract structured article data.
    """

    if not html:
        return None

    # Limit HTML size to avoid token overflow
    max_len = 40000
    content_to_send = html[:max_len] if truncate and len(html) > max_len else html

    system_prompt = f"""
You are an expert web scraping agent.

Your task is to extract the following fields from the HTML page:

1. title
2. articleUrl (absolute URL)
3. imageUrl
4. excerpt

Instructions:
{instructions}

Return ONLY a JSON object that matches the expected schema.
Do not include explanations or text outside JSON.
"""

    max_retries = 3
    for attempt in range(1, max_retries + 1):
        try:
            # The OpenAI Python client here provides a synchronous ChatCompletion
            # object; call it in a thread to avoid "can't be used in 'await'"
            completion = await asyncio.to_thread(
                client.chat.completions.create,
                model="gpt-4o-mini",
                temperature=0.1,
                messages=[
                    {"role": "system", "content": system_prompt},
                    {"role": "user", "content": content_to_send},
                ],
            )

            # Extract response content; it may be a string or a dict depending on the client
            response_content = completion.choices[0].message.content

            # If the SDK returned a dict-like object, use it directly
            if not isinstance(response_content, str):
                data = response_content
            else:
                # Clean up typical LLM wrappers like triple-backticks and markdown
                body = response_content.strip()
                m = re.search(r'```(?:json)?\s*(.*?)```', body, re.DOTALL | re.IGNORECASE)
                if m:
                    body = m.group(1).strip()

                # Find the first JSON object/array start
                first_obj = None
                for ch in ('{', '['):
                    idx = body.find(ch)
                    if idx != -1:
                        first_obj = idx
                        break
                if first_obj is not None:
                    body = body[first_obj:]

                # Try to parse JSON. If it fails, try to heuristically extract a JSON-like block.
                try:
                    data = json.loads(body)
                except Exception:
                    # Fallback: try to grab the largest {...} or [...] block
                    m2 = re.search(r'(\{.*\}|\[.*\])', body, re.DOTALL)
                    if m2:
                        try:
                            data = json.loads(m2.group(1))
                        except Exception:
                            print(f"[llm_extractor] Failed to parse JSON after fallback. Raw response:\n{response_content}")
                            raise
                    else:
                        print(f"[llm_extractor] No JSON block found in response. Raw response:\n{response_content}")
                        raise

            # Validate against Pydantic schema using parsed data
            return EuronArticleList.model_validate(data)

        except Exception as e:
            print(f"❌ Error in LLM extraction (attempt {attempt}/{max_retries}): {repr(e)}")
            traceback.print_exc()
            if attempt == max_retries:
                return None
            await asyncio.sleep(2 ** attempt)
